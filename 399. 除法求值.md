[399. 除法求值](https://leetcode-cn.com/problems/evaluate-division/)
```
给出方程式 A / B = k, 其中 A 和 B 均为代表字符串的变量， k 是一个浮点型数字。根据已知方程式求解问题，并返回计算结果。如果结果不存在，则返回 -1.0。
示例 :
给定 a / b = 2.0, b / c = 3.0
问题: a / c = ?, b / a = ?, a / e = ?, a / a = ?, x / x = ? 
返回 [6.0, 0.5, -1.0, 1.0, -1.0 ]
输入为: vector<pair<string, string>> equations, vector<double>& values, vector<pair<string, string>> queries(方程式，方程式结果，问题方程式)， 其中 equations.size() == values.size()，即方程式的长度与方程式结果长度相等（程式与结果一一对应），并且结果值均为正数。以上为方程式的描述。 返回vector<double>类型。
基于上述例子，输入如下：
equations(方程式) = [ ["a", "b"], ["b", "c"] ],
values(方程式结果) = [2.0, 3.0],
queries(问题方程式) = [ ["a", "c"], ["b", "a"], ["a", "e"], ["a", "a"], ["x", "x"] ]. 
输入总是有效的。你可以假设除法运算中不会出现除数为0的情况，且不存在任何矛盾的结果。
```
# **思路**
```
方法一：dfs
建立查找表，从查找表中str1中对应的list中依次查找下去，经过多次查找后如果能查到inter_str到str2的连接，则求得值；
设置visited，如果查找过程中到了之前经过的点，则说明str1和str2没有互联，返回-1，避免死循环；
每次查找到有效值时，若查找表中不存在str1与str2的直接关系，则添加到查找表中。
方法二：并查集（无压缩版）
构建并查集：a->b->c: (表示a/b，b/c，c/c)
要求b/a，则用(b/c) / (a/c)
```

//方法一：
class Solution {
public:
    double dfs(unordered_map< string, unordered_map<string, double> > m, string cur, string end, set<string> visited, bool &found)
    {
        if(m.find(cur)==m.end())
        {
            found=false;
            return -1.0;
        }
        if(visited.find(cur)!=visited.end())
        {
            found=false;
            return -1.0;
        }
        if(m[cur].find(end) != m[cur].end())
        {
            found=true;
            return m[cur][end];
        }
        visited.insert(cur);
        for(auto it:m[cur])
        {
            double res=dfs(m,it.first,end,visited,found);
            if(found)
            {
                res *= it.second;
                return res;
            }
        }
        return -1.0;
    }
    vector<double> calcEquation(vector<vector<string>>& equations, vector<double>& values, vector<vector<string>>& queries) {
        unordered_map< string, unordered_map<string, double> > m;
        for(int i=0; i<values.size(); i++)
        {
            string s1=equations[i][0];
            string s2=equations[i][1];
            double v=values[i];
            m[s1][s2]=v;
            m[s2][s1]=1.0/v;
        }
        vector<double> res;
        for(int i=0; i<queries.size(); i++)
        {
            string s1=queries[i][0];
            string s2=queries[i][1];
            if (m.find(s1)==m.end() || m.find(s2)==m.end())
                res.push_back(-1.0);
            else if(s1==s2)
                res.push_back(1.0);
            else
            {                
                bool found=false;
                set<string> visited;
                double tmp = dfs(m,s1,s2,visited,found);
                res.push_back(tmp);                
                if (found && m[s1].find(s2)==m[s1].end())
                {
                    m[s1][s2]=tmp;
                    m[s2][s1]=1.0/tmp;
                }
            }

        }
        return res;
    }
};
//方法二：并查集
class Solution {
public:
    unordered_map<string, string> parent;
    unordered_map<string, double> value;
    void merge(string s1, string s2, double v)
    {
        double tmp=1.0;
        while(parent[s1]!=s1)
        {
            tmp*=value[s1];
            s1=parent[s1];            
        }            
        parent[s1]=s2;
        value[s1]=v/tmp;
    }
    double find_value(string s1, string s2)
    {
        double d1=1.0;
        double d2=1.0;
        while(parent[s1]!=s1)
        {
            d1*=value[s1];
            s1=parent[s1];
        }
        while(parent[s2]!=s2)
        {
            d2*=value[s2];
            s2=parent[s2];            
        }
        if(s1!=s2)
            return -1.0;
        else
            return d1/d2;
    }
    vector<double> calcEquation(vector<vector<string>>& equations, vector<double>& values, vector<vector<string>>& queries) {
        vector<double> res;
        for(int i=0; i<equations.size(); i++)
        {
            string s1 = equations[i][0];
            string s2 = equations[i][1];
            if(parent.find(s1)==parent.end())
            {
                parent[s1]=s1;
                value[s1]=1.0;
            }
            if(parent.find(s2)==parent.end())
            {
                parent[s2]=s2;
                value[s2]=1.0;
            }
            merge(s1,s2,values[i]);
        }
        for (int i = 0; i < queries.size(); i++)
        {
            string s1=queries[i][0];
            string s2=queries[i][1];
            if (parent.find(s1)==parent.end() || parent.find(s2)==parent.end())
                res.push_back(-1.0);
            else if(s1==s2)
                res.push_back(1.0);
            else
            {
                double d = find_value(s1,s2);
                res.push_back(d);
            }
        }
        return res;
    }
};
